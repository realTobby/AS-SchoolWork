1. Aufgabe: Ändern Sie den Text des Buttons mit JQuery auf "Stressed" und stellen Sie die die Hintergrundfarbe auf "yellow" ein. Wenn der Button gedrückt wird, soll der Text des Buttons rückwärts geschrieben werden. Verwenden Sie dafür eine Schleife und die Substring Methode."

...

<script>
   var tmpText = ""; $("#btn1").attr('value', 'Stressed'); $("#btn1").css('background-color', 'yellow'); $( "#btn1" ).click(function() { var currentText = $(this).attr("value"); for(var i = currentText.length; i > -1; i--) { tmpText = tmpText + currentText.charAt(i); } $(this).attr("value", tmpText); tmpText = ""; });
</script>
